Example JS Markup:

<script src="js/jquery.mixitup.min.js"></script><!-- MixItUp plugin - for sortable portfolio -->

<!-- Initialize MixItUp plugin -->
<script>
$(function () {
var filterList = {
init: function () {

$('#projects').mixitup({
targetSelector: '.projects',
filterSelector: '.filter',
effects: ['fade'],
easing: 'snap',
// call the hover effect
onMixEnd: filterList.hoverEffect()
});

},

hoverEffect: function () {
// Hover Effect
$('#projects .projects').hover(
function () {
$(this).find('.label').stop().animate({opacity: 0.8}, 200, 'easeOutQuad');
$(this).find('.label-text h3').stop().animate({top: 0}, 200, 'easeOutQuad');
},
function () {
$(this).find('.label').stop().animate({opacity: 0}, 200, 'easeInQuad');
$(this).find('.label-text h3').stop().animate({top: '20px'}, 200, 'easeInQuad');
}
);
}

};
filterList.init();
});
</script>

1.04: Portfolio Item

Owl Carousel/Slider

The portfolio slider is produced via the Owl Carousel plugin. The html marup below will produce the core structure of the slider, and you should set your JS settings in the footer of your page, as seen below in the JS example.

Example slider HTML markup:

<link href="css/owl.carousel.css" rel="stylesheet"><!-- Call the owl carousel css file in the head of your page -->

<section class="container-fluid slider">
<div class="row">
<div class="col-xs-12 nopadding">
<div id="slider">
<div class="item"><img src="img/port-carousel/slide-1.jpg" alt="Slide One"></div>
<div class="item"><img src="img/port-carousel/slide-2.jpg" alt="Slide Two"></div>
<div class="item"><img src="img/port-carousel/slide-3.jpg" alt="Slide Three"></div>
<div class="item"><img src="img/port-carousel/slide-4.jpg" alt="Slide Four"></div>
<div class="item"><img src="img/port-carousel/slide-5.jpg" alt="Slide Four"></div>
<div class="item"><img src="img/port-carousel/slide-6.jpg" alt="Slide Four"></div>
</div>
</div>
</div>
</section>

Example slider JS markup:

<script src="js/owl.carousel.min.js"></script><!-- Call the owl carousel js file in the footer of your page -->

<!-- Initialize Owl carousel/slider with desired settings - see http://owlgraphic.com/owlcarousel/#how-to for more options -->
<script>
$("#slider").owlCarousel({
autoPlay : false,
stopOnHover : true, //Set AutoPlay to 3 seconds
items : 2,
itemsDesktop: [1199,2],
itemsDesktopSmall: [979,2],
itemsTablet: [768,2],
itemsMobile: [479,1],
rewindNav : false,
navigation: true,
pagination: false,
autoHeight: false,
navigationText : ['<span class="icon-arrow-left2"></span>','<span class="icon-arrow-right2"></span>']
});
</script>
